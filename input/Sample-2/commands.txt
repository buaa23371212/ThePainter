# 2. 绘制屋顶（三角形）

# 3. 绘制门（矩形）

# 4. 绘制左窗户
polygon -file input/Sample-2/shapes.json -id window_left

# 5. 绘制右窗户
polygon -file input/Sample-2/shapes.json -id window_right
